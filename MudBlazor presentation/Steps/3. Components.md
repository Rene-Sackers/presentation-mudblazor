## Snackbar

```
// Basic item
Snackbar.Add("What's up .NET zuid!!!", Severity.Error);

// Require interaction & Action on close
Snackbar.Add("What's up .NET zuid!!!", Severity.Error,
	o =>
	{
		o.RequireInteraction = true;
		o.CloseButtonClickFunc = _ =>
		{
			Snackbar.Add("Ok sorry.");
			return Task.CompletedTask;
		};
	});

// Render an entire page/component inside a snackbar item
Snackbar.Add<Home>();
```
